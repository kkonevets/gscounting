114 auto CSR::random(size_t nrows, size_t ncols, float prob) -> CSR {
115  std::vector<float> data;
116  std::vector<std::uint32_t> indices;
117  std::vector<std::uint32_t> indptr{0};
118 
119  auto seed(std::chrono::steady_clock::now().time_since_epoch().count());
120  std::mt19937 rng(seed);
121 
122  auto threshold = static_cast<unsigned int>(prob * 100);
123  size_t iptr = 0;
124  for (size_t _i = 0; _i < nrows; ++_i) {
125  for (size_t j = 0; j < ncols; ++j) {
126  auto r = rng();
127  if (r % 100 < threshold) {
128  auto v = static_cast<float>(r) / static_cast<float>(rng.max());
129  data.push_back(v);
130  indices.push_back(j);
131  iptr++;
132  }
133  }
134  indptr.push_back(iptr);
135  }
136  return CSR(std::move(data), std::move(indices), std::move(indptr));
137 }

146  std::fill(begin, end, 0);

148  auto worker = [&](const tbb::blocked_range<size_t> &r) {
149  for (auto i = r.begin(); i != r.end(); ++i) {
150  auto ixi{ixs[i]};
151  if (ixi < 0) {

155  if (ix >= _nrows) {

159  }
160  auto _i = i * _ncols;

163  }
164  }
