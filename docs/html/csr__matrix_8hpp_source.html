89 bool CSR::check_csr() {

92  return false;

100  return false;
101  }
102  if (data.size() != indices.size()) {
103  std::cerr << "indices and data arrays should have same size" << std::endl;
104  return false;
105  }
106  if (indptr.back() > indices.size()) {
107  std::cerr << "Last value of index pointer should be less than "

126 Dense CSR::slice(const std::vector<std::uint32_t> &ixs) {
127  assert(nrows != 0 && ncols != 0);
128  Dense d(ixs.size(), ncols);
129 
130  auto worker = [&](const tbb::blocked_range<size_t> &r) {
131  for (auto i = r.begin(); i != r.end(); ++i) {
132  size_t ix = ixs[i];
133  if (ix > nrows) {
134  fprintf(stderr, "Index %zu is out of range (0, %zu)\n", ix, nrows);
135  return;
136  }
137  auto _i = i * ncols;
138  for (size_t j = indptr[ix]; j < indptr[ix + 1]; ++j) {
139  d.data[_i + indices[j]] = data[j];
140  }
141  }
142  };
143 
144  parallel_for(tbb::blocked_range<size_t>(0, ixs.size()), worker);
145  return d;
